environment {
        AWS_CREDENTIALS = credentials('aws-credentials') // Reference to the stored AWS credentials
        TF_VAR_aws_access_key = "${AWS_CREDENTIALS_USR}"
        TF_VAR_aws_secret_key = "${AWS_CREDENTIALS_PSW}"
    }

    stages {
        stage('Checkout') {
            steps {
                // Checkout the repository
                git branch: 'main', url: 'https://github.com/Mahidhar-Mahi/Terraform-EC2-Deployment.git'
            }
        }
        stage('Terraform Init') {
            steps {
                // Initialize Terraform
                sh 'terraform init'
            }
        }
        stage('Terraform Plan') {
            steps {
                // Plan Terraform changes
                sh 'terraform plan'
            }
        }
        stage('Terraform Apply') {
            steps {
                // Apply Terraform changes
                sh 'terraform apply -auto-approve'
            }
        }
    }
    post {
        always {
            // Clean up the workspace
            cleanWs()
        }
    }
}
